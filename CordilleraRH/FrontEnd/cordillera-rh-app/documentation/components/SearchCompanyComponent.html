import { Component, OnDestroy, OnInit, ViewChild } from '@angular/core';
import { NgForm } from '@angular/forms';
import { Router } from '@angular/router';
import { CompanyModel } from './company.model';
import { CompanyService } from './company.service';

@Component({
  selector: 'app-search-company',
  templateUrl: './search.component.html',
  styleUrls: ['./search.component.css']
})
export class SearchCompanyComponent implements OnInit, OnDestroy {

  isLoading = true;

  showAllDetails = false;
  showUpdate = false;
  showTermination = false;

  @ViewChild('searchCompany') 
  searchCompany?: NgForm;
  
  sendCompanyId: number;
  ListOfCompaniees: CompanyModel[];
  errorAlert = null;

  constructor(
    private companyService: CompanyService,
    private router: Router
  ) { }

  ngOnInit() {

    setTimeout(
      () => {
        this.isLoading = true
      }, 2000
    );

    this.ListOfCompaniees = this.companyService.getAllCompanys();
    this.isLoading = false;
    this.showAllDetails = false;
  }

  ngOnDestroy(){
    this.ListOfCompaniees = null;
    this.sendCompanyId = null;
    this.errorAlert = null;
  }

  getFullDetails(id: number){
    this.sendCompanyId = id;
    this.showAllDetails = true;
  }

  getUpdateCompany(id: number){
    this.companyService.updateId = id;
    this.showUpdate = true;
  }

  onTerminateCompany(id: number){
    this.sendCompanyId = id;
    this.showTermination = true;
  }

  onSearchCompany(){

    // segunda validação
    if(this.searchCompany.value.companyName === "" && this.searchCompany.value.state === ""){
      return;
    }

    // valida os campos para pesquisar
    if(this.searchCompany.value.state === "" && this.searchCompany.value.companyName !== ""){
      // função caso apenas o nome da compania esteja preenchido
      
    }else{
      // função caso os dois estejam preenchidos

    }
  }

}
